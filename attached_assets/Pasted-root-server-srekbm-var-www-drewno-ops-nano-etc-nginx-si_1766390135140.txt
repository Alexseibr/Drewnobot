#!/bin/bash

# Backup original

# Create new config
cat > /etc/nginx/sites-enabled/d.drewno.by << 'NGINX'
server {
    server_name d.drewno.by;

    location / {
        proxy_pass http://127.0.0.1:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;

        # Disable caching for HTML
        add_header Cache-Control "no-cache, no-store, must-revalidate";
        add_header Pragma "no-cache";
        add_header Expires 0;
    }

    listen 443 ssl;
    ssl_certificate /etc/letsencrypt/live/d.drewno.by/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/d.drewno.by/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;
}

server {
    if ($host = d.drewno.by) {
        return 301 https://$host$request_uri;
    }

    listen 80;    server_name d.drewno.by;
    return 404;
}
chmod +x /tmp/fix_nginx.sh && /tmp/fix_nginx.she!"
